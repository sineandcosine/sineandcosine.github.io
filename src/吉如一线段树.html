                  核心思想：pushtag 和 modifyMin的分类操作

                  主体思想：

                  考虑对一段区间进行对x取min的操作，那么操作的对象只有大于x的元素。区间操作可以使用lazy标记下传完成的条件：

                  1. 懒标记的可合并性

                  2. 区间整体普通属性可利用懒标记和其他相关普通属性进行快速计算更新

                  而显然若对所有大于x的元素平等得判断，则无法满足第二个性质，因为无法维护大于所有变量x的所有元素与x的差值的和。那么如果用暴力的方法就必须递归到底，对叶子结点进行更新，再向上递归合并pushup进行区间更新，但显然这样虽然保证了正确性但由于是区间操作，一次操作的复杂度就是O(nlogn)。吉如一线段树巧妙地仅考虑对一种方便进行维护的情况进行lazy快速区间操作，其他情况则暂不处理而向下递归直到满足方便处理的情况。根据势能理论，这样进行维护的复杂度巧妙地为O(nlogn).

                  处理步骤：

                  对当前区间对x取min操作的情况有如下情况：

                  1. 区间mx已经小于等于x：此时无需继续设置lazy或向下递归等，直接退出即可

                  2. 区间mx大于x，但区间次大值semx严格小于x，即此时只有mx会被降低至x，并且此时区间最大仍然是x，区间最大的数量不变，这是一种非常好计算更新区间结点信息的情况，同时其子节点也必然和该节点的情况相同或最大值小于等于x，因此直接对其更新并附上lazy标记，退出即可。

                  3. x小于等于区间次大值semx，此时不方便使用lazy进行区间更新，故需要继续向下递归知道合适为止。

                  • 显然随着向下递归，节点情况只会趋向与2和1两种好维护的情况。叶子结点必然是12两种情况，设置叶子结点无次大值。因此遇到不方便区间操作的情况3可以继续向下递归是正确的。

                  对于复杂度的感性理解：

                  对于支持区间取and的数组，可以为每个区间设置历史取and为0过的位，若此次操作的值使得为0的位是之前操作总和的子集，则不必继续向下更新。而若需要向下更新，则暴力向下递归即可，可能直到叶子结点，然后pushup回来，换句话说也就是不需要lazy标记进行区间更新，因为每个元素被取and并且数值上有所更新的次数不超过30次。而在区间最值操作的问题上则不能这样，因为取min是无穷无尽的，可以不断向下取min取操作次数次。但是可以发现向下递归的过程如果按照x≤se的规则就会发现最大的至少2个值被合并为最大值。而总元素个数最多为数组长度个，那么递归合并最多为数组长度次，即区间内不同的值的个数越来越少。此时操作2可适应的区间会越来越多，即可以使用lazy的区间越来越多，所以总向下递归次数不会太多，以后就都使用lazy标记进行区间操作了。所以复杂度可以维持在O(nlogn)。

                  对于query操作使用经典的合并求值即可

                          segtrr[i].sum = segtrr[lch].sum + segtrr[rch].sum;

                      }

                          segtrr[i].sum += (tag - segtrr[i].mx) * segtrr[i].cntmx;
                          segtrr[i].mx = segtrr[i].lazy = tag;
                      }
                      inline void pushdown(int i) {
                          if (segtrr[i].lazy != -1) {
                              if (segtrr[i].l == segtrr[i].r) {
                                  segtrr[i].lazy = -1;
                                  return;
                              }

                              pushtag(i << 1 | 1, segtrr[i].lazy);
                              segtrr[i].lazy = -1;
                          }
                      }
                      inline void build(int i, int l, int r) {
                          segtrr[i].l = l, segtrr[i].r = r;
                          if (l == r) {
                              segtrr[i].mx = segtrr[i].sum = arr[l];
                              segtrr[i].cntmx = 1;
                              return;
                          }
                          int mid = l + r >> 1;
                          build(i << 1, l, mid);
                          build(i << 1 | 1, mid + 1, r);
                          pushup(i);
                      }
                      inline void modifyMin(int i, int l, int r, int val) {
                          if (segtrr[i].r < l || r < segtrr[i].l || segtrr[i].mx <= val) return;

                              return pushtag(i, val);
                          }
                          pushdown(i);
                          modifyMin(i << 1, l, r, val);
                          modifyMin(i << 1 | 1, l, r, val);
                          pushup(i);
                      }
                      inline int queryMax(int i, int l, int r) {
                          pushdown(i);
                          if (segtrr[i].r < l || segtrr[i].l > r) return -INF;
                          if (l <= segtrr[i].l && segtrr[i].r <= r) return segtrr[i].mx;
                          return max(queryMax(i << 1, l, r), queryMax(i << 1 | 1, l, r));
                      }
                      inline int querySum(int i, int l, int r) {
                          pushdown(i);
                          if (segtrr[i].r < l || r < segtrr[i].l) return 0;

                      }

                  例题：大小最值的lazy的互相影响：

                  BZOJ-4695. 最假女选手

                  题解：

                  对于取min更新还要对取max标记、mi属性和semi进行取小操作，取max同理。总体还是直到方便处理的只有最值需要改动时才进行lazy标记。

                  总代码：

                  xxxxxxxxxx
                  #include <bits/stdc++.h>
                  

                  using namespace std;
                  #define int long long 
                  #define INF 1e17
                  #define endl '\n'
                  

                  class SegmentBeats {
                  public:
                      class node {
                      public:
                          int l, r;
                          int mx = -INF, semx = -INF, cntmx = 0, tgmx = -INF;
                          int mi = INF, semi = INF, cntmi = 0, tgmi = INF;
                          int sum = 0, tgsm = 0;
                      };
                      SegmentBeats(vector<int> arr) {
                          this->n = arr.size() - 1;
                          this->arr = arr;
                          this->segtrr.resize(n << 2);
                          build(1, 1, n);
                      }
                  

                      void pushup(int i) {
                          int lch = (i << 1), rch = (i << 1 | 1);
                          segtrr[i].sum = segtrr[lch].sum + segtrr[rch].sum;
                          if (segtrr[lch].mx == segtrr[rch].mx) {
                              segtrr[i].mx = segtrr[lch].mx;
                              segtrr[i].cntmx = segtrr[lch].cntmx + segtrr[rch].cntmx;
                              segtrr[i].semx = max(segtrr[lch].semx, segtrr[rch].semx);
                          } else if (segtrr[lch].mx > segtrr[rch].mx) {
                              segtrr[i].mx = segtrr[lch].mx;
                              segtrr[i].cntmx = segtrr[lch].cntmx;
                              segtrr[i].semx = max(segtrr[lch].semx, segtrr[rch].mx);
                          } else {
                              segtrr[i].mx = segtrr[rch].mx;
                              segtrr[i].cntmx = segtrr[rch].cntmx;
                              segtrr[i].semx = max(segtrr[lch].mx, segtrr[rch].semx);
                          }
                  

                          if (segtrr[lch].mi == segtrr[rch].mi) {
                              segtrr[i].mi = segtrr[lch].mi;
                              segtrr[i].cntmi = segtrr[lch].cntmi + segtrr[rch].cntmi;
                              segtrr[i].semi = min(segtrr[lch].semi, segtrr[rch].semi);
                          } else if (segtrr[lch].mi < segtrr[rch].mi) {
                              segtrr[i].mi = segtrr[lch].mi;
                              segtrr[i].cntmi = segtrr[lch].cntmi;
                              segtrr[i].semi = min(segtrr[lch].semi, segtrr[rch].mi);
                          } else {
                              segtrr[i].mi = segtrr[rch].mi;
                              segtrr[i].cntmi = segtrr[rch].cntmi;
                              segtrr[i].semi = min(segtrr[lch].mi, segtrr[rch].semi);
                          }
                      }
                  

                      void build(int i, int l, int r) {
                          segtrr[i].l = l, segtrr[i].r = r;
                          if (l == r) {
                              segtrr[i].mx = segtrr[i].mi = segtrr[i].sum = arr[l];
                              segtrr[i].cntmx = segtrr[i].cntmi = 1;
                              return;
                          }
                          int mid = l + r >> 1;
                          build(i << 1, l, mid);
                          build(i << 1 | 1, mid + 1, r);
                          pushup(i);
                      }
                      void pushAdd(int i, int tag) {
                          if (segtrr[i].mx != -INF) segtrr[i].mx += tag;
                          if (segtrr[i].mi != INF) segtrr[i].mi += tag;
                          if (segtrr[i].tgmx != -INF) segtrr[i].tgmx += tag;
                          if (segtrr[i].tgmi != INF) segtrr[i].tgmi += tag;
                          if (segtrr[i].semx != -INF) segtrr[i].semx += tag;
                          if (segtrr[i].semi != INF) segtrr[i].semi += tag;
                          segtrr[i].sum += (segtrr[i].r - segtrr[i].l + 1) * tag;
                          segtrr[i].tgsm += tag;
                      }
                      void pushMin(int i, int tag) {
                          if (segtrr[i].mx <= tag) return;
                          segtrr[i].sum += (tag - segtrr[i].mx) * segtrr[i].cntmx;
                          segtrr[i].mx = segtrr[i].tgmi = tag;
                          segtrr[i].mi = min(segtrr[i].mi, tag);
                          if (segtrr[i].semi != INF) segtrr[i].semi = min(segtrr[i].semi, tag);
                          segtrr[i].tgmx = min(segtrr[i].tgmx, tag);
                      }
                      void pushMax(int i, int tag) {
                          if (segtrr[i].mi >= tag) return;
                          segtrr[i].sum += (tag - segtrr[i].mi) * segtrr[i].cntmi;
                          segtrr[i].mi = segtrr[i].tgmx = tag;
                          segtrr[i].mx = max(segtrr[i].mx, tag);
                          if (segtrr[i].semx != -INF) segtrr[i].semx = max(segtrr[i].semx, tag);
                          segtrr[i].tgmi = max(segtrr[i].tgmi, tag);
                      }
                      
                      void pushdown(int i) {
                          if (segtrr[i].l == segtrr[i].r) {
                              segtrr[i].tgsm = 0;
                              segtrr[i].tgmx = -INF;
                              segtrr[i].tgmi = INF;
                              return;
                          }
                          if (segtrr[i].tgsm != 0) {
                              pushAdd(i << 1, segtrr[i].tgsm);
                              pushAdd(i << 1 | 1, segtrr[i].tgsm);
                              segtrr[i].tgsm = 0;
                          }
                          if (segtrr[i].tgmx != -INF) {
                              pushMax(i << 1, segtrr[i].tgmx);
                              pushMax(i << 1 | 1, segtrr[i].tgmx);
                              segtrr[i].tgmx = -INF;
                          }
                          if (segtrr[i].tgmi != INF) {
                              pushMin(i << 1, segtrr[i].tgmi);
                              pushMin(i << 1 | 1, segtrr[i].tgmi);
                              segtrr[i].tgmi = INF;
                          }
                  

                      }
                      void modifyMin(int i, int l, int r, int val) {
                          if (segtrr[i].r < l || r < segtrr[i].l || segtrr[i].mx <= val) return;
                          if (l <= segtrr[i].l && segtrr[i].r <= r && segtrr[i].semx < val) return pushMin(i, val);
                          pushdown(i);
                          modifyMin(i << 1, l, r, val);
                          modifyMin(i << 1 | 1, l, r, val);
                          pushup(i);
                      }
                      void modifyMax(int i, int l, int r, int val) {
                          if (segtrr[i].r < l || r < segtrr[i].l || segtrr[i].mi >= val) return;
                          if (l <= segtrr[i].l && segtrr[i].r <= r && segtrr[i].semi > val) return pushMax(i, val);
                          pushdown(i);
                          modifyMax(i << 1, l, r, val);
                          modifyMax(i << 1 | 1, l, r, val);
                          pushup(i);
                      }
                      void rangeAdd(int i, int l, int r, int val) {
                          if (segtrr[i].r < l || r < segtrr[i].l) return;
                          if (l <= segtrr[i].l && segtrr[i].r <= r) return pushAdd(i, val);
                          pushdown(i);
                          rangeAdd(i << 1, l, r, val);
                          rangeAdd(i << 1 | 1, l, r, val);
                          pushup(i);
                      }
                      int queryMax(int i, int l, int r) {
                          pushdown(i);
                          if (segtrr[i].r < l || r < segtrr[i].l) return -INF;
                          if (l <= segtrr[i].l && segtrr[i].r <= r) return segtrr[i].mx;
                          return max(queryMax(i << 1, l, r), queryMax(i << 1 | 1, l, r));
                      }
                      int queryMin(int i, int l, int r) {
                          pushdown(i);
                          if (segtrr[i].r < l || r < segtrr[i].l) return INF;
                          if (l <= segtrr[i].l && segtrr[i].r <= r) return segtrr[i].mi;
                          return min(queryMin(i << 1, l, r), queryMin(i << 1 | 1, l, r));
                      }
                      int querySum(int i, int l, int r) {
                          pushdown(i);
                          if (segtrr[i].r < l || r < segtrr[i].l) return 0;
                          if (l <= segtrr[i].l && segtrr[i].r <= r) return segtrr[i].sum;
                          return querySum(i << 1, l, r) + querySum(i << 1 | 1, l, r);
                      }

                      vector<node> segtrr;

                  2.历史最值问题

                  CPU 监控（板子例题）

                  题意：

                  维护一个数组支持种操作：

                  1. 区间所有加 x

                  2. 区间覆盖为 x

                  3. 询问区间当前 max

                  4. 询问区间历史 max

                  题解：

                  线段树结点信息：

                  xxxxxxxxxx
                  class node {
                  public:
                      int l, r;
                      int mx = -INF, mxmx = -INF;     //区间当前最大值，历史最大值
                      int add = 0, addmx = 0;         //区间加的lazy，区间历史最大加的lazy
                      int st = -INF, stmx = -INF;     //区间覆盖值lazy，区间历史最大覆盖值lazy
                  };

                  • 注意：普通属性只有题目要求的mx和mxmx，其他均为lazy标记用于下传！

                  pushup:

                  xxxxxxxxxx
                  void pushup(int i) {
                      segtrr[i].mx = max(segtrr[i << 1].mx, segtrr[i << 1 | 1].mx);
                      segtrr[i].mxmx = max(segtrr[i << 1].mxmx, segtrr[i << 1 | 1].mxmx);
                  }

                  经典pushup，只对普通属性（除lazy外的属性）pushup。

                  思想核心:pushdown

                  主体思想：

                  1. 首先不考虑区间覆盖操作，仅考虑区间加操作，由于同时有对当时和历史最大值的查询需求，故对于区间加的lazy标记要记录当时add和历史最大add两项，记为add和addmx。那么对于标记下传，add的下传很容易，直接让子节点add加上父节点的add值，得到的就是下传标记这时候的子节点对于区间的加值。而对于segtrr[son].addmx来说，应当让其和segtrr[son].add + segtrr[father].addmx取min，这是因为segtrr[son].add是上一次标记下传到son时候更新给它当时的区间add值，而那时候到现在下传标记之间该子节点没有被下传到过，也就是没有被更新过。同时上一次下传至该子节点势必导致父节点的lazy标记被清空，所以现在segtrr[father].addmx是在该子节点上次更新以后的某时刻到现在的、father对应区间的历史最大add值。那么son可以选择使用上次下传标记至它之前得到的历史最大add值，也就是现在的segtrr[son].addmx，或者使用在那之后到现在这段时间内的历史最大add值，而这个值的计算就是子节点最后更新的add值和它对于区间的、从上次更新到现在的历史最大add值之和，也就是segtrr[son].add + segtrr[father].addmx。同时，这也证明了标记的可下传性，即父节点的lazy更新总是在子节点之后，且由于下传后及时清空lazy导致父节点的lazy对应时间段的开始时间严格在子节点lazy对应时间段的最终时间之后。

                    而对于mx和mxmx的更新则也是类似，对于当前最大值mx，直接加上当前区间加值add，而历史最大值则使用子节点最终更新前的历史最大值，或使用最终时刻最大值加上后续时间段中获得的历史最大区间加addmx，两者取max。

                    （这里说的结点更新均指某次pushdown的过程中pushdown到了该结点，即该结点利用了父节点的lazy标记而进行新值计算，我们就说这个点在此时进行了一次更新）

                  2. 其次考虑区间覆盖操作。这里使用了一个小技巧1，即一段区间若没有准备对子区间进行区间覆盖的操作（也就是没有曾经的区间覆盖操作卡在当前结点未下传，或说所有历史覆盖操作都已经下传并体现在子节点的mx和mxmx之中了）时只使用add和addmx标记进行更新，并且利用这两个标记对mx和mxmx进行更新（因为此时是对原数进行修改，原数存在mx和mxmx之中）。而若当前点有曾经的区间覆盖操作未下传，即它的st标记有值，则对于区间加操作应当是应用在st标记上，因为该add操作是在子节点的那个覆盖操作之后的应当是建立在子节点st的标记上进行下传而不是子子节点的mx和mxmx，那都是未进行覆盖操作之前的（因为覆盖操作还未下传），而同时子节点的add标记则正是对子子节点的mx和mxmx进行修改。对st标记的赋值关系和给add标记赋值完全类似。总而言之就是若有未下传的覆盖则此时就不再使用add标记了，直接对st进行更新到时候利用st标记对子子节点的mx和mxmx更新与st标记同时达到目的。

                  3. 顺序问题：显然根据2的分析，add标记的下传应当在set标记下传之前进行，若一个结点同时具有add和st标记，则有add是对其子节点的mx进行的修改，而st是对上次下传至该节点后第一次覆盖后的所有操作（包括add和st）的最值。所以add标记下传修改mx和mxmx对应的时刻应当在st标记下传覆盖之前，所以add应当先下传。

                  pushAdd：用于利用自身lazy标记add和addmx修改子节点。

                  xxxxxxxxxx

                  二者综合：

                  P6242 【模板】线段树 3

                  时刻牢记吉如一线段树的精髓：

                  • 对于最值修改维护：将最值和其他值分开考虑，修改时向下递归直到操作对象只有最值时放置tag返回。

                  • 对于历史最值维护：转移式，某属性历史最值等于其上次记录的历史最值和上次记录的最后值与上方结点下传来的，在该节点最后更新后的历史获得最值之和取最值

                  本题灵活利用了最值和其他值分开考虑。同时有区间加法和区间取min，而对于取min操作由于只对mx进行操作（若会对其他数有操作则会向下递归而避之），故对于mx来说等价于加上了(val - mx)，而对于其他数保持不变。那么可以将mx和其他值的add标记分开，取min就看做只对mx进行处理的区间加法操作，真正的区间加则是对mx和其他数就行的区间加法。

                  其他要注意的就是min操作是只对放置tag时候的mx进行的，而标记下传的过程中会逐步从2情况（semx≤val≤mx）转向1情况（mx≤val)，故每次利用标记更新子节点时要先判断最初放置tag的时候的mx是否为当前子节点对应区间的mx，若否则对于该区间的mx也和当前区间的其他值一样处理。

                  代码： 

                  xxxxxxxxxx
                  #include <bits/stdc++.h>
                  

                  using namespace std;
                  #define int long long 
                  #define endl '\n'
                  #define INF 9e18
                  

                  class SegmentBeats {
                  public:
                      class node {
                      public:
                          int l, r;
                          int sum = 0, mx = -INF, mxmx = -INF;
                          int cntmx = 0, semx = -INF;
                          /*---------lazy----------*/
                          int mxadd = 0, mxaddmx = 0;
                          int add = 0, addmx = 0;
                      };
                      SegmentBeats(vector<int> arr) {
                          this->n = arr.size() - 1;
                          this->arr = arr;
                          this->segtrr.resize(n << 2);
                          build(1, 1, n);
                      }
                      void pushup(int i) {
                          int lch = i << 1, rch = i << 1 | 1;
                          segtrr[i].sum = segtrr[lch].sum + segtrr[rch].sum;
                          segtrr[i].mxmx = max(segtrr[lch].mxmx, segtrr[rch].mxmx);
                          if (segtrr[lch].mx == segtrr[rch].mx) {
                              segtrr[i].mx = segtrr[lch].mx;
                              segtrr[i].cntmx = segtrr[lch].cntmx + segtrr[rch].cntmx;
                              segtrr[i].semx = max(segtrr[lch].mx, segtrr[rch].mx);
                          } else if (segtrr[lch].mx > segtrr[rch].mx) {
                              segtrr[i].mx = segtrr[lch].mx;
                              segtrr[i].cntmx = segtrr[lch].cntmx;
                              segtrr[i].semx = max(segtrr[lch].semx, segtrr[rch].mx);
                          } else {
                              segtrr[i].mx = segtrr[rch].mx;
                              segtrr[i].cntmx = segtrr[rch].cntmx;
                              segtrr[i].semx = max(segtrr[lch].mx, segtrr[rch].semx);
                          }
                      }
                      void build(int i, int l, int r) {
                          segtrr[i].l = l; segtrr[i].r = r;
                          if (l == r) {
                              segtrr[i].mx = segtrr[i].mxmx = segtrr[i].sum = arr[l];
                              segtrr[i].cntmx = 1;
                              return;
                          }
                          int mid = l + r >> 1;
                          build(i << 1, l, mid);
                          build(i << 1 | 1, mid + 1, r);
                          pushup(i);
                      }
                      void pushMxAdd(int i, int mxadd, int mxaddmx) {
                          segtrr[i].mxmx = max(segtrr[i].mxmx, segtrr[i].mx + mxaddmx), segtrr[i].mx += mxadd;
                          segtrr[i].sum += segtrr[i].cntmx * mxadd;
                          segtrr[i].mxaddmx = max(segtrr[i].mxaddmx, segtrr[i].mxadd + mxaddmx), segtrr[i].mxadd += mxadd;
                      }
                      void pushNomAdd(int i, int add, int addmx) {
                          segtrr[i].semx += add;
                          segtrr[i].sum += (segtrr[i].r - segtrr[i].l + 1 - segtrr[i].cntmx) * add;
                          segtrr[i].addmx = max(segtrr[i].addmx, segtrr[i].add + addmx), segtrr[i].add += add;
                      }
                      void pushdown(int i) {
                          if (segtrr[i].l == segtrr[i].r) {
                              segtrr[i].add = segtrr[i].addmx = segtrr[i].mxadd = segtrr[i].mxaddmx = 0;
                              return;
                          }
                          int lch = i << 1, rch = i << 1 | 1;
                          int mx = max(segtrr[lch].mx, segtrr[rch].mx);
                          if (mx == segtrr[lch].mx && mx == segtrr[rch].mx) {
                              pushMxAdd(lch, segtrr[i].mxadd, segtrr[i].mxaddmx);
                              pushNomAdd(lch, segtrr[i].add, segtrr[i].addmx);
                              pushMxAdd(rch, segtrr[i].mxadd, segtrr[i].mxaddmx);
                              pushNomAdd(rch, segtrr[i].add, segtrr[i].addmx);
                          } else if (mx == segtrr[lch].mx) {
                              pushMxAdd(lch, segtrr[i].mxadd, segtrr[i].mxaddmx);
                              pushNomAdd(lch, segtrr[i].add, segtrr[i].addmx);
                              pushMxAdd(rch, segtrr[i].add, segtrr[i].addmx);
                              pushNomAdd(rch, segtrr[i].add, segtrr[i].addmx);
                          } else {
                              pushMxAdd(lch, segtrr[i].add, segtrr[i].addmx);
                              pushNomAdd(lch, segtrr[i].add, segtrr[i].addmx);
                              pushMxAdd(rch, segtrr[i].mxadd, segtrr[i].mxaddmx);
                              pushNomAdd(rch, segtrr[i].add, segtrr[i].addmx);
                          }
                          segtrr[i].add = segtrr[i].addmx = segtrr[i].mxadd = segtrr[i].mxaddmx = 0;
                      }
                      void rangeAdd(int i, int l, int r, int val) {
                          pushdown(i);
                          if (segtrr[i].r < l || r < segtrr[i].l) return;
                          if (l <= segtrr[i].l && segtrr[i].r <= r) {
                              pushMxAdd(i, val, max(val, 0LL));
                              pushNomAdd(i, val, max(val, 0LL));
                              return;
                          }
                          rangeAdd(i << 1, l, r, val);
                          rangeAdd(i << 1 | 1, l, r, val);
                          pushup(i);
                      }
                      void modifyMin(int i, int l, int r, int val) {
                          pushdown(i);
                          if (segtrr[i].r < l || r < segtrr[i].l || segtrr[i].mx <= val) return;
                          if (l <= segtrr[i].l && segtrr[i].r <= r && segtrr[i].semx < val) {
                              return pushMxAdd(i, val - segtrr[i].mx, 0);
                          }
                          modifyMin(i << 1, l, r, val);
                          modifyMin(i << 1 | 1, l, r, val);
                          pushup(i);
                      }
                      int querySum(int i, int l, int r) {
                          pushdown(i);
                          if (segtrr[i].r < l || r < segtrr[i].l) return 0;
                          if (l <= segtrr[i].l && segtrr[i].r <= r) return segtrr[i].sum;
                          return querySum(i << 1, l, r) + querySum(i << 1 | 1, l, r);
                      }
                      int queryMax(int i, int l, int r) {
                          pushdown(i);
                          if (segtrr[i].r < l || r < segtrr[i].l) return -INF;
                          if (l <= segtrr[i].l && segtrr[i].r <= r) return segtrr[i].mx;
                          return max(queryMax(i << 1, l, r), queryMax(i << 1 | 1, l, r));
                      }
                      int queryMaxMax(int i, int l, int r) {
                          pushdown(i);
                          if (segtrr[i].r < l || r < segtrr[i].l) return -INF;
                          if (l <= segtrr[i].l && segtrr[i].r <= r) return segtrr[i].mxmx;
                          return max(queryMaxMax(i << 1, l, r), queryMaxMax(i << 1 | 1, l, r));
                      }
                      
                      int n;
                      vector<int> arr;
                      vector<node> segtrr;
                  };

                  1 可能会有的疑问：为什么不先下传st，这样先进行结点覆盖再进行区间加就不用考虑覆盖操作后的add标记被st标记覆盖了，也就不需要判断若子节点有st标记则将add下传到子节点的st标记中了。但这会产生一个严重的问题：若一个结点同时有add标记和st标记待下传，而add标记是在得到st标记之前得到的，即add标记是对下层覆盖操作前进行下传的。那么这个add标记是对子节点历史mx和mxmx进行修改的，若st先进行覆盖则这个add标记就失效了。简单说标记下传涉及三方面：一次覆盖操作前的加操作，覆盖操作和覆盖后的加操作，add标记对应覆盖前的加操作，初始的st标记对应覆盖操作，而后判断若下方结点有st标记则add标记下传到st标记之中则对应于覆盖后的加操作。 ↩
